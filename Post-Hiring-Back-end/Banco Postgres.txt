CREATE SCHEMA PUBLIC AUTHORIZATION postgres;

CREATE TABLE PUBLIC.SECTOR(
	SECTOR_ID SERIAL4,
	SECTOR_NAME VARCHAR(255),
	DESCRIPTION VARCHAR(255),
	PRIMARY KEY (SECTOR_ID)
);

CREATE TABLE PUBLIC.DOCUMENTS(
	DOCUMENT_ID SERIAL4,
	TITLE VARCHAR(255),
	MATERIAL_liNK VARCHAR(255),
	DOCUMENT_CONTENT VARCHAR(500),
	SECTOR_ID INT4,
	PRIMARY KEY (DOCUMENT_ID),
	FOREIGN KEY (SECTOR_ID) REFERENCES SECTOR(SECTOR_ID)
);

CREATE TABLE PUBLIC.USER(
	USER_ID SERIAL4,
	USER_NAME VARCHAR(255),
	BIRTH_DATE DATE,
	SECTOR_ID INT4,
	IS_SUPERVISOR BOOLEAN,
	NICKNAME VARCHAR(255),
	EMAIL VARCHAR(255),
	STATUS BOOLEAN DEFAULT TRUE,
	PASSWORD VARCHAR(255),
	PRIMARY KEY (USER_ID),
	FOREIGN KEY (SECTOR_ID) REFERENCES SECTOR(SECTOR_ID)   
);

CREATE TABLE PUBLIC.USER_ADM(
	USER_ID INT4,
	USER_ADM_ID SERIAL4,
	PRIMARY KEY (USER_ADM_ID),
	FOREIGN KEY (USER_ID) REFERENCES PUBLIC.USER(USER_ID)  
);

CREATE TABLE PUBLIC.USER_ASSOCIATE(
	USER_ID INT4,
	SUPERVISOR_ID INT4,
	USER_ASSOCIATE_ID SERIAL4,
	PRIMARY KEY (USER_ASSOCIATE_ID),
	FOREIGN KEY (USER_ID) REFERENCES PUBLIC.USER(USER_ID),
	FOREIGN KEY (SUPERVISOR_ID) REFERENCES PUBLIC.USER_ADM(USER_ADM_ID) 
);
